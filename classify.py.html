import cohere

co = cohere.Client(API_KEY)


class classify:
    def __init__(self):
        self.co = cohere.Client(API_KEY)

    def introduction(self, input):
        input = int(input)
        age = "baby"
        if input <= 9:
            age = "Kid"
        elif input <= 12:
            age = "Child"
        elif input <= 15:
            age = "Young Teen"
        else:
            age = "Teen"
        return age

    def categorize(self, input):
        theQuestion = input
        questionType = "null"

        classifications = self.co.classify(
            model='small',
            inputs=[theQuestion],
            examples=[Example("I just got my period", "physical question"),
                      Example("I do not want to be seen as a boy", "emotional question"),
                      Example("Why do my classmates bully me", "emotional question"),
                      Example("I have a lot of mood swings and don't know what to do", "emotional question"),
                      Example("I am angry", "emotional question"),
                      Example("I believe in God", "spiritual question"),
                      Example("I am thinking of exploring Islam", "spiritual question"),
                      Example("What if I don't believe in God", "spiritual question"),
                      Example("Do all religions lead to the same destination", "spiritual question"),
                      Example("What happens after I die", "spiritual question"),
                      Example("I have a crush on my best friend", "romantic question"),
                      Example("I think I am a lesbian", "romantic question"),
                      Example("I identify as asexual", "romantic question"),
                      Example("I dont want to participate in hookup culture anymore", "romantic question"),
                      Example("I am having problems with my partner", "romantic question"),
                      Example("I am insecure about the size of my genitals", "physical question"),
                      Example("Why is my body changing so much", "physical question"),
                      Example("How do I use a condom", "physical question"),
                      Example("I want to go on birth control", "physical question"),
                      Example("I struggle with anxiety", "emotional question")])

        for classification in classifications.classifications:
            if classification.prediction == "emotional question":
                questionType = "Emotional"
            elif classification.prediction == "spiritual question":
                questionType = "Spiritual"
            elif classification.prediction == "romantic question":
                questionType = "Romantic"
            elif classification.prediction == "physical question":
                questionType = "Physical"
        return questionType
